#include <iostream>

using namespace std;

int main()
{
   int iter = 0;
   string word1, word2 = "SAM";
   getline(cin, word1);
   for (int i = 0; i < word1.length();i++)
   {
      if (word1[i] == word2[iter])
      {
         iter++;
         if (word1[i] == 'A')
            cout << i << ' ';
         if (word1[i] == 'M')
            cout << i + 1;
      }
   }
}
