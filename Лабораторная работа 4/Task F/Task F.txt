
#include <iostream>
using namespace std;

int main(){
   string word;
   int n_c = 0;
   getline(cin, word);
   int letter_count[26] = {0};
   string answer;
   char medium_letter = '\0';
   for (char c: word) letter_count[c - 'A']++;
   for (int i = 0; i < 26; i++){
      if (letter_count[i] % 2 == 1){
         if (n_c > 0){
            cout << "IMPOSSIBLE";
            return 0;
         }
         n_c++;
         medium_letter = 'A' + i;
      }
      for (int j = 0; j < letter_count[i] / 2; j++) answer += 'A' + i;
   }
   if (n_c == 1)
      cout << answer << medium_letter;
   else
      cout << answer;
   for (int i = answer.length() - 1; i >= 0; i--) cout << answer[i];
}
